// AIArticleGenerator.tsx

import React, { useState } from 'react';

import { articleAPI } from '../services/article-api';

interface Props {
  mode: 'create' | 'update';
  articleId?: number;  // Obrigat�rio se mode === 'update'
  onSuccess: (article: Article) => void;
}

const AIArticleGenerator: React.FC<Props> = ({ mode, articleId, onSuccess }) => {
  const [request, setRequest] = useState<AIArticleRequest>({
    articleId: articleId,
    prompt: '',
    generateImage: false,
    categoryId: undefined,
    roles: []
  });
  const [loading, setLoading] = useState(false);

    setError(null);

    try {
      let result: Article;

      }
      
      onSuccess(result);
    } catch (err) {
      setError(err.message || 'Erro ao gerar artigo com IA');
    } finally {
      setLoading(false);
    }
  };

  return (
    <div className="ai-article-generator">

      </button>
